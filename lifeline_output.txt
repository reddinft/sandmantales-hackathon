  "Accept": "audio/mpeg",
  "Content-Type": "application/json",
  "xi-api-key": "YOUR_API_KEY"
}

  "text": "Bonjour, comment Ã§a va aujourd'hui?",  # French example

    "similarity_boost": 0.80,
    "style": 0.0,
    "use_speaker_boost": True
  }
}
